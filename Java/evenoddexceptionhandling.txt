public class ParityChecker {

    // 'throws' warns that this method might reject your input
    public static void checkParity(int number) throws IllegalArgumentException {
        
        if (number < 0) {
            // 'throw' is the action that stops the code if the number is negative
            throw new IllegalArgumentException("Negative numbers are not allowed!");
        }

        if (number % 2 == 0) {
            System.out.println(number + " is Even.");
        } else {
            System.out.println(number + " is Odd.");
        }
    }

    public static void main(String[] args) {
        try {
            checkParity(10);  // Works fine
            checkParity(-5);  // This will trigger the 'throw'
        } catch (IllegalArgumentException e) {
            System.out.println("Caught an exception: " + e.getMessage());
        }
    }
}
